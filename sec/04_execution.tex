%--------------------------------------------------------------------
\subsection{Phase 0: Project scaffolding}

Create the source tree and install dependencies.

\smallskip\noindent\textbf{Layout.}\quad
\texttt{src/\{config, data\_loader, baseline, speculative, metrics, evaluate, visualize, utils\}.py}; directories \texttt{manifests/}, \texttt{results/}, \texttt{figures/}.

\smallskip\noindent\textbf{Dependencies.}\quad
\texttt{torch}, \texttt{transformers}, \texttt{datasets}, \texttt{accelerate}, \texttt{rouge-score}, \texttt{pandas}, \texttt{matplotlib}, \texttt{seaborn}.

\smallskip\noindent\textbf{Tasks.}
\begin{enumerate}\itemsep0pt
    \item Install packages into \texttt{.venv}.
    \item Write \texttt{config.py} (model IDs, hyperparameters, prompt templates, paths).
    \item Write \texttt{utils.py} (seed setting, GPU timer, CSV logging).
    \item Freeze \texttt{requirements.txt} and record environment in \texttt{env\_spec.txt}.
\end{enumerate}

\noindent\textbf{Deliverable:} All imports resolve; torch device available.

%--------------------------------------------------------------------
\subsection{Phase 1: Data preparation}

Load, sample, and freeze evaluation subsets.

\smallskip\noindent\textbf{Tasks.}
\begin{enumerate}\itemsep0pt
    \item Load GSM8K (test, $n{=}300$), MMLU ($5{\times}100$), CNN/DailyMail (test, $n{=}200$) with \texttt{seed=42}.
    \item Save frozen sample-ID manifests to \texttt{manifests/*.json}.
    \item Verify tokenizer compatibility across all Qwen~2.5 models.
    \item 20-sample smoke test on Qwen2.5-7B.
\end{enumerate}

\noindent\textbf{Deliverable:} Three manifest JSONs; tokenizer check passed; smoke-test log.

%--------------------------------------------------------------------
\subsection{Phase 2: Baseline benchmarking}

Run target-only autoregressive decoding on all 1{,}000 samples in both regimes.

\smallskip\noindent\textbf{Tasks.}
\begin{enumerate}\itemsep0pt
    \item Load Qwen2.5-7B-Instruct in fp16, \texttt{device\_map="auto"}.
    \item Record per sample: \texttt{sample\_id}, \texttt{task}, latency, TTFT, TPOT, token count, output text.
    \item Run deterministic ($T{=}0$) and stochastic ($T{=}0.7$) regimes.
    \item Compute baseline quality (exact-match, letter-match, ROUGE-L).
    \item Save to \texttt{results/baseline\_\{det,stoch\}.csv}.
\end{enumerate}

\noindent\textbf{Deliverable:} Two CSVs; baseline quality scores.

%--------------------------------------------------------------------
\subsection{Phase 3: AR speculative grid A\,---\,0.5B draft}

Speculative decoding with Qwen2.5-0.5B-Instruct as autoregressive draft~\cite{Leviathan2023}.

\smallskip\noindent\textbf{Tasks.}
\begin{enumerate}\itemsep0pt
    \item Implement the AR speculative loop: draft generates $k$ tokens sequentially; target verifies in one forward pass; accept prefix, reject and resample at divergence.
    \item Run $k \in \{4,8,16\} \times \{\text{det},\text{stoch}\} = 6$ configs.
    \item Compute $\alpha$, $B_{\text{eff}}$, $S$, and quality metrics per config.
    \item Save to \texttt{results/spec\_0.5B\_k\{4,8,16\}\_\{det,stoch\}.csv}.
\end{enumerate}

\noindent\textbf{Deliverable:} Six CSVs with acceptance and speedup per config.

%--------------------------------------------------------------------
\subsection{Phase 4: AR speculative grid B\,---\,1.5B draft}

Repeat Phase~3 with Qwen2.5-1.5B-Instruct; add stability analysis.

\smallskip\noindent\textbf{Tasks.}
\begin{enumerate}\itemsep0pt
    \item Run the same 6 configs with the 1.5B draft.
    \item Identify top-2 configs by $S$ (subject to $|\Delta Q| \leq 1.0$).
    \item Re-run those with seeds $\{42, 123, 999\}$ to obtain $\sigma_S$.
    \item Save to \texttt{results/spec\_1.5B\_*} and \texttt{results/stability/}.
\end{enumerate}

\noindent\textbf{Deliverable:} Six CSVs; stability analysis; full $12{\times}12$ AR results matrix.

%--------------------------------------------------------------------
\subsection{Phase 5: Quality and error analysis}

Systematic comparison of all speculative outputs against baseline.

\smallskip\noindent\textbf{Tasks.}
\begin{enumerate}\itemsep0pt
    \item Compute $\Delta Q = Q_{\text{spec}} - Q_{\text{base}}$ for every configuration.
    \item Identify disagreement samples; bucket by output length.
    \item Analyse rejection clustering by token position (early vs.\ late).
    \item Generate error taxonomy: truncation, substitution, semantic drift.
\end{enumerate}

\noindent\textbf{Deliverable:} Quality-delta table; disagreement analysis; error taxonomy.

%--------------------------------------------------------------------
\subsection{Phase 6: Synthesis, visualization, and paper}

Generate all figures and tables; complete remaining sections.

\smallskip\noindent\textbf{Figures.}
\begin{enumerate}\itemsep0pt
    \item Speedup $S$ vs.\ $\Delta Q$ Pareto frontier (colour by draft model).
    \item Acceptance rate $\alpha$ vs.\ $k$, grouped by draft model size.
    \item $\alpha$ vs.\ temperature (deterministic vs.\ stochastic).
\end{enumerate}

\smallskip\noindent\textbf{Tables.}
\begin{enumerate}\itemsep0pt
    \item Full results matrix: all 12 speculative configs $\times$ all metrics.
    \item Quality comparison: baseline vs.\ best speculative per task.
\end{enumerate}

\smallskip\noindent\textbf{Paper tasks.}
\begin{enumerate}\itemsep0pt
    \item Write \texttt{sec/01b\_related\_work.tex}: Leviathan et al.~\cite{Leviathan2023}, Chen et al.~\cite{Chen2023specsampling} (speculative sampling), Stern et al.~\cite{Stern2018blockwise} (blockwise parallel), Medusa~\cite{Cai2024Medusa}, EAGLE~\cite{Li2024Eagle,Li2024Eagle2}.
    \item Write \texttt{sec/06\_results.tex}: narrative answering RQ1--RQ3.
    \item Fill abstract placeholders \texttt{[X]}, \texttt{[Y]}, \texttt{[Z]}.
    \item Update conclusion with actual findings.
\end{enumerate}

\noindent\textbf{Deliverable:} Submission-ready manuscript with all figures, tables, and sections.
